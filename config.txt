default: &default
  adapter: postgresql
  encoding: unicode
  pool: <%= ENV.fetch("RAILS_MAX_THREADS") { 5 } %>

  timeout: 5000

development:
  <<: *default
  database: hermes_development
  username: <%= ENV['POSTGRES_USER'] %>

test:
  <<: *default
  database: hermes_test
  username: <%= ENV['POSTGRES_USER'] %>
  password: <%= ENV['POSTGRES_PASSWORD'] %>

  <<: *default
  database: <%= ENV['POSTGRES_DB'] %>
